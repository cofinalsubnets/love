cmake_minimum_required(VERSION 3.13)

# Route every compile through ccache when it's on PATH. Set before project()
# so it covers the Pico SDK's own TUs as well as gwen's. ASM too: the SDK has a
# fair amount of .S, and ccache handles the assembler fine.
find_program(CCACHE_PROGRAM ccache)
if(CCACHE_PROGRAM)
  set(CMAKE_C_COMPILER_LAUNCHER   ${CCACHE_PROGRAM})
  set(CMAKE_CXX_COMPILER_LAUNCHER ${CCACHE_PROGRAM})
  set(CMAKE_ASM_COMPILER_LAUNCHER ${CCACHE_PROGRAM})
endif()

include(pico_sdk_import.cmake)

project(main C CXX ASM)

pico_sdk_init()

# gwen core: every g/*.c translation unit (g.c was split per-concern, so a
# single-file build no longer links). CONFIGURE_DEPENDS so a newly added g/*.c
# (e.g. when g.c was split out) re-triggers the glob at build time instead of
# silently dropping out and failing the link with undefined references.
file(GLOB G_SOURCES CONFIGURE_DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/../core/*.c)
# vt.c (the data self-quote sentinel TU) is compiled on its own below so its
# section strides can be reflected into a generated vt.h; drop it from the
# main glob to avoid linking the sentinels twice.
list(REMOVE_ITEM G_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/../core/vt.c)

# boot.h / repl.h are the lcat-embedded prelude, generated by the host
# build (it owns lcat). Regenerate them via the host Makefile whenever the
# .g sources change; they land in ../host/b.
set(GEN_DIR ${CMAKE_CURRENT_SOURCE_DIR}/../host/b)
add_custom_command(
  OUTPUT ${GEN_DIR}/prelude.h ${GEN_DIR}/repl.h
  COMMAND make -C ${CMAKE_CURRENT_SOURCE_DIR}/../host b/prelude.h b/repl.h
  DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/../core/prelude.g
          ${CMAKE_CURRENT_SOURCE_DIR}/../core/repl.g
  COMMENT "Generating prelude.h / repl.h via host lcat")
add_custom_target(gl_headers DEPENDS ${GEN_DIR}/prelude.h ${GEN_DIR}/repl.h)

# The data-sentinel TU on its own. It bootstraps from the portable
# core/vt.h (it never calls g_typ, so the stride it records is independent of
# which header it sees) and inherits the SDK's global arch flags + include dirs.
add_library(gdata OBJECT ${CMAKE_CURRENT_SOURCE_DIR}/../core/vt.c)
add_dependencies(gdata gl_headers)
# Only the include dirs -- NOT target_link_libraries(pico_stdlib ...), which on
# an OBJECT library would fold every SDK object-library source into gdata. core/
# is portable (no pico/ or hardware/ headers) and the arch flags are global, so
# the include dirs plus the SDK toolchain's global CFLAGS are all vt.c needs.
target_include_directories(gdata PRIVATE ../core ../host/b)

# Reflect the per-sentinel byte stride out of gdata's object and bake it into a
# generated vt.h whose g_typ() drops the portable header's two link-time
# divides. On Thumb the 24-byte body is 16-aligned, so the stride is 32 and the
# index-divide becomes a shift by 5; gen_data_vt.py warns if it isn't a power of
# two. main's include path puts ${VT_DIR} ahead of ../core so it shadows the
# portable core/vt.h.
set(VT_DIR ${CMAKE_CURRENT_BINARY_DIR}/gen)
set(GEN_VT ${CMAKE_CURRENT_SOURCE_DIR}/../tools/gen_data_vt.py)
add_custom_command(
  OUTPUT ${VT_DIR}/vt.h
  COMMAND ${CMAKE_COMMAND} -E make_directory ${VT_DIR}
  COMMAND ${GEN_VT} $<TARGET_OBJECTS:gdata> -o ${VT_DIR}/vt.h
  DEPENDS $<TARGET_OBJECTS:gdata> ${GEN_VT}
  COMMENT "Reflecting data_vt stride into ${VT_DIR}/vt.h"
  VERBATIM)
add_custom_target(gl_data_vt DEPENDS ${VT_DIR}/vt.h)

add_executable(main
    main.c
    ${G_SOURCES}
    $<TARGET_OBJECTS:gdata>
)
add_dependencies(main gl_headers gl_data_vt)
# ${VT_DIR} BEFORE ../core so the generated vt.h shadows the portable one.
target_include_directories(main BEFORE PRIVATE ${VT_DIR})
target_include_directories(main PRIVATE ../core ../host/b)

# memmap_stack.ld relocates the core-0 stack out of the cramped 4 KB
# SCRATCH_Y bank into a region at the top of main SRAM. We keep it modest
# (8 KB) so the rest of SRAM is heap -- gwen's copying GC needs room for the
# transient old+new pool peak as it grows the pool along the Fibonacci
# sequence. PICO_STACK_SIZE must match the STACK region length in the script.
pico_set_linker_script(main ${CMAKE_CURRENT_SOURCE_DIR}/memmap_stack.ld)
target_compile_definitions(main PRIVATE PICO_STACK_SIZE=0x2000)

# Disable the threaded-code tail-call dispatch (g_tco) on the RP2040; fall
# back to the trampolined Continue()/Ap() loop.
target_compile_definitions(main PRIVATE g_tco=0)

# pico_malloc's wrapper panics ("Out of memory") on allocation failure by
# default, which kills the board before USB even enumerates. gwen's g_ini_m
# / GC are written to handle a NULL return (they GC-and-retry, then report
# g_status_oom), so turn the panic off and let malloc fail the normal way.
target_compile_definitions(main PRIVATE PICO_MALLOC_PANIC=0)

# USB CDC console only. UART stdio is off so GPIO0/GPIO1 stay free GPIOs
# (a UART TX pin idles high, which is why GPIO0 read as "on" before).
pico_enable_stdio_usb(main 1)
pico_enable_stdio_uart(main 0)

pico_add_extra_outputs(main)

target_link_libraries(main pico_stdlib hardware_gpio)
